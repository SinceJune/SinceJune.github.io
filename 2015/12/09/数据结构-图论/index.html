图的表示

　　图有许多的表示方法，常见的有邻接矩阵，邻接表，邻接多重表和十字链表表示法。下面将简略的介绍一下每一种表示方法。

邻接矩阵

　　邻接矩阵使用横坐标和纵坐标来表示顶点，交叉的值来表示是否相通，是一种十分容易理解的方法。

邻接表

　　邻接表与邻接矩阵的思想相似，在邻接表中每个顶点有一个链表，后面跟着从这个顶点出去的所有边的集合，分有向表和无向表。

邻接多重表

　　邻接多重表也分有向表和无向表，但对链表边的结构进行了更改。

　　其中iVex和jVex是与某条边依附的两个顶点在顶点表中的下标。iLink指向依附顶点iVex的下一条边，jLink指向依附顶点jVex的下一条边。

　　也就是说在邻接多重表里边，边表存放的是一条边，而不是一个顶点。

十字链表

　　十字链表的好处就是因为把邻接表和逆邻接表整合在了一起，这样既容易找到以Vi为尾的弧，也容易找到以Vi为头的弧，因而容易求得顶点的出度和入度。

　　十字链表除了结构复杂一点外，其实创建图算法的时间复杂度是和邻接表相同的，因此，在有向图的应用中，十字链表也是非常好的数据结构模型。

　　十字链表更改了表的顶点的结构，

　　同时也变更了表的边的结构。

搜索

　　说到图，自然最常用的操作就是图的搜索，其中深度搜索和广度搜索首当其冲，以下例子使用图的邻接表表示方法，来展示图的深度搜索和广度搜索方案。

DFS(深度搜索)

void  dfs(FILE* fp, int  v)
{
    nodePointer  w;
    visited[v] = 1;
    fprintf(fp, "V%d ", v);
    for (w = graph[v]; w; w = w->next)
        if (!visited[w->vertex])
            dfs(fp, w->vertex);

}

　　深度搜索类似于二叉树的先序遍历，先找到一个顶点的第一个相邻顶点，再对这个顶点寻找其相邻的第一个顶点，循环往复，直到所有可遍历到的顶点都遍历完。

BFS(广度搜索)

void  bfs(int  v)
{
    nodePointer w;
    fprintf(fp, "V%d ", v);
    visited[v] = 1;
    addq(v);
    while (!IsEmpty())
    {
        v = deleteq();
        for (w = graph[v]; w; w = w->next)
        if (!visited[w->vertex])
        {
            fprintf(fp, "V%d ", w->vertex);
            addq(w->vertex);
            visited[w->vertex] = 1;
        }
    }
}

　　广度搜索使用了队列，类似于二叉树的层级遍历(levelOrder)的方法，将一个顶点的相邻顶点全部入队，每出队一个元素，便将其临界点入队，直到所有可遍历到的顶点都遍历完。

　　主代码段：


#include < stdio.h>
#include "PublicDS.h"
#define  MAX_VERTICES  50  
typedef  struct  node  *nodePointer;
typedef  struct  node   {
    int    vertex;
    int    edge;
    struct  node    *next;
};
int queue[MAX_VERTICES];
nodePointer graph[MAX_VERTICES];
int  n = 0;  
int visited[MAX_VERTICES] = { 0 };
int real = -1;
int font = -1;
void addq(int n)
{
    if (real == MAX_VERTICES - 1)
    {
        printf("queue full!\n");
        return;
    }
    queue[++real] = n;
    return;
}
int deleteq()
{
    if (real == font)
        return 0;
    return queue[++font];
}
int IsEmpty()
{
    if (real == font)
        return 1;
    else return 0;
}

void  dfs(FILE* fp, int  v)
{
    nodePointer  w;
    visited[v] = 1;
    fprintf(fp, "V%d ", v);
    for (w = graph[v]; w; w = w->next)
        if (!visited[w->vertex])
            dfs(fp, w->vertex);

}
void  bfs(FILE* fp, int  v)
{
    nodePointer w;
    fprintf(fp, "V%d ", v);
    visited[v] = 1;
    addq(v);
    while (!IsEmpty())  {
        v = deleteq();
        for (w = graph[v]; w; w = w->next)
            if (!visited[w->vertex])  {
                fprintf(fp, "V%d ", w->vertex);
                addq(w->vertex);
                visited[w->vertex] = 1;
            }
    }
}

int main()
{
    FILE* fp = fopen("input.txt", "r");
    if (fp == NULL)
    {
        fprintf(stderr, "file open failed!\n");
        exit(1);
    }
    fscanf(fp, "%d", &n);
    int m = 0;

    //申请空间 
    for (int i = 0; i < n; ++i)
    {
        MALLOC(graph[i], sizeof(struct node), nodePointer);
        graph[i]->next = NULL;
    }
    fscanf(fp, "%d", &m);

    //添加边
    int temp;
    nodePointer tmp;
    while (m > 0)
    {
        fscanf(fp, "%d  ", &temp);

        //插入到空节点之后
        tmp = graph[temp];
        while (tmp->next != NULL)
            tmp = tmp->next;
        MALLOC(tmp->next, sizeof(struct node), nodePointer);
        tmp->next->next = NULL;

        //第一个结点为头结点
        fscanf(fp, "%d  %d", &tmp->next->vertex, &tmp->next->edge);

        --m;
    }

    //去掉头结点
    for (int i = 0; i < n; ++i)
    {
        tmp = graph[i]->next;
        free(graph[i]);
        graph[i] = tmp;
    }

    //定义输出文件
    FILE* fout = fopen("output.txt", "w");
    if (fout == NULL)
    {
        fprintf(stderr, "file open failed!\n");
        exit(1);
    }

    //输入
    int choice;
    while (1)
    {
        fscanf(fp, "%d", &choice);
        if (choice == -1)
            break;
        fprintf(fout, "BFS From V%d:", choice);
        bfs(fout, choice);
        fprintf(fout, "\n");

        //恢复每一个结点的visited为0
        for (int i = 0; i < n; i++)
            visited[i] = 0;
    }
    fclose(fp);
    fclose(fout);
    return 0;
}
### 最短路径 　　图的经典最短路径算法有两种，第一种是迪杰斯特拉算法 ，另一种是佛罗伊德算法。第一种实现从某一顶点到另一顶点的最短路径计算，第二种算法则实现任意两顶点最短路径的计算，是加强版的迪杰斯特拉算法。 　　其核心思想，就是从起始点A出发，找到其最接近的顶点B，再从这个顶点B出发，寻找周围的顶点C，如果A-B和B-C的距离之和小于A-C的直接距离，就用这个距离和替代A-C的距离。 　　示例：

nodePointer graph[MAX_VERTICES];
int mdistance[MAX_VERTICES] = { 1000,1000,1000,1000,1000,10000 };
short int mfound[MAX_VERTICES] = { 0 };
int  n = 0;

void OutShortestPath(int distance[], int iStartV, int n);
void  shortestpath(int  v, int  distance[], int  n, short  int found[]);
void  shortestpath(int  v, int  distance[], int  n, short  int  found[])
{
    int  i, u;
    for (i = 0; i < n; i++)
        found[i] = FALSE;
    for (nodePointer t = graph[v]; t; t = t->next)
        distance[t->vertex] = t->edge;
    found[v] = TRUE;                            //v访问
    distance[v] = 0;                            //v到v的距离为0
    for (i = 0; i < n - 2; i++)
    {
        u = choose(distance, n, found);   /* O( n ) */

        found[u] = TRUE;

        for (nodePointer t = graph[u]; t; t = t->next)
        {
            if (!found[t->vertex])
            {
                if (distance[u] + t->edge < distance[t->vertex])
                    distance[t->vertex] = distance[u] + t->edge;
            }
        }
    }   /* end for-loop */
}
